Ticket 4.2: Asynchronous Domain Routing
Goal: Create a factory that inspects the file, spins up a background thread to parse it, and safely returns it to the main UI thread.

Step 1: UI State Management
In your main application class (Program.cs or MainWindow.cs), we need to prevent the app from drawing the highway while a song is being parsed.

Add a boolean flag: private bool _isLoading = false;

Add a string: private string _loadingMessage = "";

Step 2: The Song Factory Method
Create an asynchronous method to handle the routing. This method determines which parser to use based on the extension.

C#
private async Task LoadSongAsync(string filePath) {
    _isLoading = true;
    _loadingMessage = $"Loading {Path.GetFileName(filePath)}...";

    // 1. Clean up existing song and stop audio
    _currentSong?.Dispose();
    _currentSong = null;

    string ext = Path.GetExtension(filePath).ToLower();

    try {
        // 2. Offload the heavy parsing to a background thread
        _currentSong = await Task.Run(async () => {
            if (ext == ".mid" || ext == ".midi") {
                _loadingMessage = "Parsing MIDI...";
                return new PianoSong(filePath); 
            } 
            else if (ext == ".psarc") {
                _loadingMessage = "Extracting CDLC and decoding audio (this may take a moment)...";
                var parser = new RocksmithParser();
                return await parser.ParsePsarcAsync(filePath);
            }
            throw new NotSupportedException("Unknown file format.");
        });
    }
    catch (Exception ex) {
        // Store exception message to show in an ImGui error modal
        Console.WriteLine($"Failed to load song: {ex.Message}");
    }
    finally {
        _isLoading = false;
    }
}
Step 3: Update the Main Render Loop
Update your ImGui render frame to check the loading state before trying to call _currentSong.Draw().

public void RenderFrame(float currentAudioTimeMs) {
    ImGui.Begin("OmniRhythm Engine");
    
    if (_isLoading) {
        // Draw a loading spinner or text
        ImGui.SetCursorPosY(ImGui.GetWindowHeight() / 2);
        ImGui.SetCursorPosX(ImGui.GetWindowWidth() / 2 - 100);
        ImGui.TextColored(new Vector4(1, 1, 0, 1), _loadingMessage);
    } 
    else if (_currentSong != null) {
        ImDrawListPtr drawList = ImGui.GetWindowDrawList();
        _currentSong.Update(currentAudioTimeMs);
        _currentSong.Draw(drawList);
    } 
    else {
        ImGui.Text("Drag and drop a .mid or .psarc file here.");
    }
    
    ImGui.End();
}
